# Function to get unique values for any column
def get_unique_values(dataframe, column_name):
    """Return unique values and count for a specified column."""
    if column_name not in dataframe.columns:
        print(f"Column '{column_name}' not found in dataset.")
        return None

    unique_vals = dataframe[column_name].dropna().unique()
    unique_count = len(unique_vals)

    print(f"\nColumn: '{column_name}'")
    print(f"  Unique values: {unique_count:,}")
    print(f"  Missing values: {dataframe[column_name].isna().sum():,}")

    # Show sample of unique values (first 10)
    if unique_count > 0:
        print(f"  Sample values (up to 10):")
        for val in list(unique_vals)[:10]:
            print(f"    - {val}")

    return unique_vals
